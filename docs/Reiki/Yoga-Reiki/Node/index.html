Module Yoga-Reiki.Node

A Yoga layout node.

This is the base of the package, as it's the unit of composition of layouts.

type ctx = {
width : int,
height : int,
direction : direction,
};
type computed_layout = {
left : int,
top : int,
width : int,
height : int,
};
type attrs = {
width : int,
height : int,
justify_content : justify_content,
flex_direction : flex_direction,
align_items : alignment,
};
type t;

Abstract node type. Kept abstract to prevent access to internal library shenanigans.

let create : attrs => t;
let insertChild : t => int => t => t;
let calculateLayout : ctx => t => t;
let computedLayout : t => computed_layout;
